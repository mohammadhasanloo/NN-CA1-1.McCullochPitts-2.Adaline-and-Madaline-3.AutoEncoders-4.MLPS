class AdalineModel:

    def __init__(self, learning_rate=0.01):
        self.alpha = learning_rate
        self.w = np.zeros(X.shape[1])
        self.b = 0        
        
    def fit(self, X, y, epochs):
      loss_history = []

      for _ in range(epochs):
          for j in range(X.shape[0]):
              # Forward each data
              ypj = np.dot(X[j], self.w) + self.b

              # Loss for each data
              error = y[j] - ypj
              self.w += self.alpha * error * X[j].T
              self.b += self.alpha * error

          # Forward
          yp = np.dot(X, self.w) + self.b

          # Loss
          errors = y - yp
          loss = (errors ** 2).sum() / 2
          loss_history.append(loss)

      return loss_history

    def predict(self, X):
        yp = np.dot(X, self.w) + self.b
        return 2*(yp >= 0).astype(np.int32) - 1

class Madaline:

    def __init__(self, hidden_units, learning_rate=0.9):
        self.alpha = learning_rate
        self.hidden_units = hidden_units

        self.w = (np.random.rand(X.shape[1], self.hidden_units) - 0.5) / 2
        self.b1 = (np.random.rand(self.hidden_units) - 0.5) / 2

        self.v = np.ones(self.hidden_units) / self.hidden_units
        self.b2 = (self.hidden_units - 1) / self.hidden_units

        self.epochs = 0
        self.max_epochs = 300

    def fit(self, X, y):

        loss_history = []

        while True:
            is_weight_changed = False
            for j in range(X.shape[0]):

                # Forward each data
                z_in = np.dot(X[j], self.w) + self.b1
                z = self.activation(z_in)
                yp_in = np.dot(z, self.v) + self.b2
                yp = self.activation(yp_in)

                if y[j] == 1 and yp == -1:
                    is_weight_changed = True
                    k = np.argmin(np.abs(z_in))
                    self.b1[k] += self.alpha * (1 - z_in[k])
                    self.w[:, k] += self.alpha * (1 - z_in[k]) * X[j].T

                elif y[j] == -1 and yp == 1:
                    is_weight_changed = True
                    K = np.where(0 < z_in)[0]
                    for k in K:
                        self.b1[k] += self.alpha * (-1 - z_in[k])
                        self.w[:, k] += self.alpha * (-1 - z_in[k]) * X[j].T

            # Forward
            yp = self.predict(X)

            # Loss
            errors = y - yp
            loss = (errors ** 2).sum() / 2
            loss_history.append(loss)
            self.epochs += 1
            if (not is_weight_changed) or (self.max_epochs == self.epochs):
                break

        return loss_history

    def activation(self, X):
        return 2*(X >= 0).astype(np.int32) - 1

    def predict(self, X):
        z_in = np.dot(X, self.w) + self.b1
        z = self.activation(z_in)
        yp_in = np.dot(z, self.v) + self.b2
        yp = self.activation(yp_in)
        return yp

def plot_distribution(df, clf, hidden_units):
  t = np.linspace(-1.5, 1.5, 2)
  decision_line = []

  for i in range(hidden_units):
    decision_line.append(-(clf.w[0][i] * t + clf.b1[i])/clf.w[1][i])

  fig, ax = plt.subplots(figsize=(5, 4), dpi=90)
  ax.scatter(df.iloc[: , 0][df.iloc[: , -1] == 1], df.iloc[: , 1][df.iloc[: , -1] == 1])
  ax.scatter(df.iloc[: , 0][df.iloc[: , -1] == 0], df.iloc[: , 1][df.iloc[: , -1] == 0])
  for i in range(hidden_units):
    ax.plot(t, decision_line[i], color='tab:red')
  fig.tight_layout()
  plt.xlim([-2, 2])
  plt.ylim([-2, 2])
  plt.show()
